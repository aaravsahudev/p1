    <script>
        let isReversed = false;
        
        function changeSpeed(speed) {
            const spinningText = document.getElementById('spinningText');
            spinningText.className = `spinning-text speed-${speed}`;
            if (isReversed) {
                spinningText.style.animationDirection = 'reverse';
            }
        }
        
        function toggleDirection() {
            const spinningText = document.getElementById('spinningText');
            isReversed = !isReversed;
            spinningText.style.animationDirection = isReversed ? 'reverse' : 'normal';
        }
    </script>
